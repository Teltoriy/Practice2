git clone -b master https://github.com/Teltoriy/Practice2  - клонирование ветки master на локалку

git status - проевка клонированного репозитория

git branch back - создание ветки back

git checkout back - переход на ветку back

git add * - добавление всех файлов в коммит

git commit -m <some message> - создание коммита

git push origin back - пуш всех коммитов на удаленный репозиторий
